13def property_filter(property_data, downpayment, mortgage_rate = None, mortgage_term = None, max_monthly_payment = None, max_loan = None): 

14 ''' Returns a dataframe containing the properties/areas. 

15  

16 Arguments 

17 ---------- 

18 data : dataframe  

19 Areas/properties in column index 0 (str) 

20 Respective prices in column index 1 (numeric)  

21  

22 downpayment : numeric 

23 Your maximal downpayment 

24  

25 mortgage_rate : numeric  

26 Interest rate on the mortgage loan (leave empty if mortgage_term is provided) 

27  

28 mortgage_term : int  

29 Contract length in years (1 to 10) for the mortgage interest rate. 

30 Only specify if you do not know what mortgage_rate to enter (leave empty if mortgage_rate provided) 

31  

32 max_monthly_payment : numeric  

33 Your max affordable or bank limited monthly payment towards your home 

34  

35 max_loan : numeric 

36 Max eligible loan based on your downpayment 

37 

38 Return 

39 ------ 

40 dataframe 

41 Properties/Areas 

42 Prices/Average area price 

43 Minimum_Downpayment 

44 Mortgage_Insurance 

45 Principal 

46 Monthly_Payment 

47 Shortest_Amortization 

48 Total_Interest 

49 Net_Cost (assuming no other fees) 

50 ''' 

51 

52 warnings.filterwarnings("ignore") 

53 

54 try: 

55 # is object a dataframe? 

56 if isinstance(property_data, pd.DataFrame) == False: 

57 raise FilterInputError('Dataframe object expected') 

58 # right number of columns? 

59 if len(property_data.columns)!=2: 

60 raise FormatError('Expected two columns of type str and numeric, respectively') 

61 # is column at index 1 (price) numeric? 

62 if pd.api.types.is_numeric_dtype(property_data.iloc[:,1]) == False: 

63 raise TypeError('Column at index 1 (price) must be numeric') 

64 

65 except FilterInputError as FIE: 

66 print(FIE, '\nReveived object of type:', type(property_data)) 

67 return None 

68 except FormatError as FE: 

69 print(FE,'\nReveived dataframe with this many columns:', len(property_data.columns)) 

70 return None 

71 except TypeError as TE: 

72 print(TE) 

73 return None 

74 

75 data = property_data.copy() 

76 

77 # Rename columns 

78 data.set_axis(['Property/Area', 'Price'], axis=1, inplace=True) 

79 

80 # Note original input of properties 

81 og_prop_count = data['Property/Area'].count() 

82 

83 # FILTER: Downpayment. Remove properties where minimal DP exceeds your entered DP 

84 data['Minimum_Downpayment'] = data.iloc[:, 1].apply(lambda x: min_downpayment(x)) 

85 data = data[data['Minimum_Downpayment'] <= downpayment] 

86 

87 # Mortgage rate. If none provided give a reasonable estimate 

88 if mortgage_rate == None: 

89 mortgage_rate = mort_rate(mortgage_term) 

90 

91 # Calculate mortgage insurance lump sum for each property 

92 default_insurance = [] 

93 for price in data['Price']: 

94 default_insurance += [mortgage_insurance(price, downpayment)] 

95 data['Mortgage_Insurance'] = default_insurance 

96 

97 # Calculate initial principal for each property by converting annual mortgage rate to a monthly rate 

98 data['Principal'] = round((data['Price'] - downpayment + data['Mortgage_Insurance']), 2) 

99 

100 # FILTER: Max eligible loan. Remove properties where the principal exceeds you max approved loan. 

101 data = data[data['Principal'] < max_loan] 

102 

103 # Add two columns for monthly payment and shortest amortization period 

104 Monthly_Payment = []; Amortization = [] 

105 for princ in data['Principal']: 

106 Monthly_Payment += [optimal_monthly_payment(princ, mortgage_rate, max_monthly_payment)[0]] 

107 Amortization += [optimal_monthly_payment(princ, mortgage_rate, max_monthly_payment)[1]] 

108 data['Monthly_Payment'] = Monthly_Payment 

109 data['Shortest_Amortization'] = Amortization 

110 

111 # FILTER: Remove rows where Monthly_Payment is null 

112 data = data[data['Monthly_Payment'].notnull()] 

113 

114 # Add column for the cumulative cost of interest given that amortization 

115 tot_int = [] 

116 for princ, monthly_payment in data[['Principal', 'Monthly_Payment']].itertuples(index=False): 

117 tot_int += [total_interest(princ, mortgage_rate, monthly_payment)] 

118 data['Total_Interest'] = tot_int 

119 

120 # Add column for net cost of home (price + cumulative interest + mortgage insurance) 

121 data['Net_Cost'] = data['Price'] + data['Mortgage_Insurance'] + data['Total_Interest'] 

122 

123 print(f"You can afford {data['Property/Area'].count()} properties from the {og_prop_count} you've provided.") 

124 return data

130def min_downpayment(price): 

131 ''' Returns the minimum downpayment required for a real estate 

132 price defined by the Financial Consumer Agency of Canada. 

133 (https://www.canada.ca/en/financial-consumer-agency/services/mortgages/down-payment.html) 

134  

135 Arguments 

136 ---------- 

137 price : numeric 

138 Property price or avereage area property price 

139  

140 Return 

141 ------ 

142 float 

143 minimum downpayment 

144 ''' 

145 try: 

146 if price < 0: 

147 print("Invalid price") 

148 return None 

149 elif price < 500000: 

150 return price*0.05 

151 elif price < 1000000: 

152 return (500000*0.05 + (price - 500000)*0.1) 

153 return price*0.2 

154 

155 except TypeError: 

156 print("Invalid price input. Must be of type numeric") 

157 return None

163def mort_rate(term): 

164 ''' If no mortgage rate is specified this function can be used to 

165 return an estimated mortgage rate based on a regression fit (R^2 = 0.926) 

166 on average Canadian mortgage rates for possible term lengths. 

167 (https://www.superbrokers.ca/tools/mortgage-rates-comparison/) 

168  

169 Arguments 

170 ---------- 

171 term : int 

172 contract length in years (from 1 to 10 years) 

173  

174 Return 

175 ------ 

176 float 

177 interest rate 

178 ''' 

179 try: 

180 if term < 1: 

181 raise TermError 

182 elif term > 10: 

183 raise TermError('Lengths greater than 10 years are not typically available.') 

184 elif isinstance(term, float): 

185 print('Warning: Term lengths are typically in whole years not fractions of years.') 

186 

187 x = term 

188 return round((0.0167*x**2 - 0.0337*x + 1.6851), 3) 

189 

190 except TermError as TE: 

191 print(f'{TE} Terms must range from (1,10) years.') 

192 return None

199def mortgage_insurance(price, downpayment): 

200 ''' Returns the cost of mortgage insurance. 

201  

202 Insurance rates are calculated from loan to asset price ratio. 

203 Rates are applied to the loan to generate a lump sum amount that's 

204 then added to the principal of the loan to give mortgage insurance. 

205  

206 Arguments 

207 ---------- 

208 price : numeric 

209 Property price 

210  

211 downpayment : int or float 

212 Downpayment on property 

213  

214 Return 

215 ------ 

216 float 

217 Mortgage insurance 

218 ''' 

219 try: 

220 DP_proportion = downpayment / price 

221 

222 # if downpayment more than 20% of the house price no mortgage insurance required. 

223 if DP_proportion >= 0.2: 

224 return 0 

225 elif DP_proportion < 0.05: 

226 raise PovertyError('Downpayment must be at least 5% the asset value') 

227 

228 loan_to_price = (price-downpayment)/price 

229 x = loan_to_price 

230 

231 # loan to price ratio determines insurance rate 

232 insurance_rate = (2924.5*x**4 - 9340.3*x**3 + 11116*x**2 - 5830.8*x + 1137.1)/100 

233 

234 # mortgage insurance is a % applied to the mortgage amount (price - downpayment) 

235 return round(((price - downpayment) * insurance_rate), 2) 

236 

237 except TypeError: 

238 print('Bad entry type. Received:', type(price), type(downpayment)) 

239 return None 

240 except PovertyError as PE: 

241 print(PE, '\nInput value is too low to be legally considered.') 

242 return None 

243 except ZeroDivisionError: 

244 print('Price cannot be zero.') 

245 return None

251def optimal_monthly_payment(principal, mortgage_rate, max_monthly_payment): 

252 ''' Returns the first amortization period which has a monthly payment 

253 less than your max_monthly_payment (ie. within budget). The shortest 

254 possible amortization period has the lowest long term interest cost. 

255 

256 Arguments 

257 ---------- 

258 principal : numeric 

259  

260 mortgage_rate : float 

261 Annual mortgage rate (loan interest) 

262  

263 max_monthly_payment: numeric 

264 Your max affordable monthly contribution 

265  

266 Return 

267 ------ 

268 list 

269 mp: monthly payment for a given amortization 

270 i: amortization period in years 

271 ''' 

272 try: 

273 for i in range(1, 26): 

274 mp = monthly_payment(principal, mortgage_rate, i, months = False) 

275 if mp <= max_monthly_payment: 

276 return [mp, i] 

277 return [np.nan, np.nan] 

278 

279 except TypeError: 

280 print('Bad entry type. Received:', type(principal), type(mortgage_rate), type(max_monthly_payment)) 

281 return None

287def monthly_payment(principal, mortgage_rate, amortization, months = False): 

288 ''' Returns the monthly payment required to meet the given amortization period. 

289 Assumes payments occur on a monthly basis. 

290 

291 Arguments 

292 ---------- 

293 principal : numeric 

294  

295 mortgage_rate : float 

296 Annual mortgage rate (loan interest) 

297  

298 amortization: int 

299 Amortization period in years (or in months if months == True) 

300  

301 months : bool  

302 (Optional) if True, amortization period is interpreted in months (default = False) 

303  

304 Return 

305 ------ 

306 float 

307 monthly payment 

308 ''' 

309 

310 R = (mortgage_rate/100/12 + 1) ## monthly interest rate 

311 

312 if months == True: 

313 n = amortization ## if specified in months, amortization = the number of payments  

314 n = amortization*12 ## convert amortization in years to the number of monthly payments 

315 

316 monthly_contribution = principal*((R**n)*(1-R) / (1-R**n)) 

317 

318 return round(monthly_contribution, 2)

324def total_interest(principal, mortgage_rate, monthly_payment): 

325 ''' Returns the cumulative interest paid on a given principal, mortgage rate, and monthly payment. 

326  

327 Arguments 

328 ---------- 

329 principal : numeric 

330  

331 mortgage_rate : float 

332 Annual mortgage rate (loan interest) 

333  

334 amortization: int 

335 Amortization period in years (or in months if months == True) 

336  

337 monthly_payment : bool  

338 Monthly contribution towards the principal 

339  

340 Return 

341 ------ 

342 float 

343 Cumulative interest paid 

344 ''' 

345 try: 

346 R = mortgage_rate/1200 ## monthly interest rate 

347 CumInterest = 0 

348 

349 i = principal * R 

350 new_p = principal + i - monthly_payment 

351 

352 while new_p > 0: 

353 CumInterest += i 

354 i = new_p * R 

355 new_p = new_p + i - monthly_payment 

356 

357 if new_p >= new_p - i + monthly_payment: 

358 print("Monthly contribution is insufficient to pay off the original Principal.") 

359 return None 

360 

361 return round(CumInterest, 2) 

362 

363 except TypeError: 

364 print('Bad entry type. Received:', type(principal), type(mortgage_rate), type(monthly_payment)) 

365 return None
